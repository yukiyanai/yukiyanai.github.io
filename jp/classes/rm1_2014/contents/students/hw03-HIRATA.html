cho_n<- function(n,k){ #nkresult
  if(n<k){             #nk
  return(0)            #result0
  }
  
  else{                #n>k
    result<- fan_n(n)/(fan_n(k)*fan_n(n-k))
    return(result)     #result
  }
}
