1.階乗を計算する関数を作りなさい

1-1.関数の定義

fct <- function(n = NULL){
  #引数はnとする(階乗する数)
  #返り値はxとする(階乗した数)
  if(n < 0){          ##n=0の時
  return(NULL)

  } else if (n == 0){ ##n=負の値をとる時
  return(1)

  } else if (n > 0){  ##n=正の値をとる時
  x <- n
  while(n > 1){
  x <- x * (n-1)    ##引数nと(n-1)をかける
  n <- n - 1        ##ループ
  }
  return(x)
  }

}

2.自分で作った階乗の関数を利用して、組み合わせを計算する関数を作りなさい

2-1.関数の定義

chs <- function(n = NULL, k = NULL){       #chs(組み合わせを計算する関数) 
  #引数はn,kとする(n個の中からkを選ぶ)
  #返り値はoutputとする(計算結果)

  output <- fct(n) / (fct(k) * fct(n - k)) #組み合わせの計算(fctは階乗を計算する関数)
  return(output)
}

3.自分で作った組み合わせの関数を利用して、二項分布の確率質量を求める関数を作りなさい

3-1.関数の定義

BinomDens <- function(x = NULL, size = NULL, prob = NULL){  #1 BinomDens(二項分布の確率質量を求める関数)
  #2 引数はx,size,probとする。それぞれ,x=パラメータ,size=試行回数,prob=成功確率である
  #3 返り値はoutput(確率質量関数)とする

  output <- rep(NA, length(x))  #4 結果を保存するベクトル

    for(i in 1:length(x)){

    output[i] <- chs(n = size, k = x[i]) * (prob ^ x[i]) * (1 - prob) ^ (size - x[i])
  } #5 計算結果を#4で作成したベクトルに代入
  return(output)
}

4.自分で作った二項分布の関数を使い、nが2,3,5,10の場合について,確率質量を図示しなさい。pの値については任意とする。ただし、2つ以上のpで図を作りなさい。その際、for()ループを少なくとも1回使いなさい。

4-1.関数の定義

BinomDens.sim <- function(n = NULL, p.vec = NULL){ #1 BinomDens.sim(シミュレーションを行う関数)
  #2 引数はnとp.vec。n=nのベクトル、p.vec=成功確率である。
  #3 返り値はoutput(確率質量関数)とする

  #4 結果を保存するベクトル
  output <- matrix(data = NA, nrow = length(0:n), ncol = (length(p.vec)) + 1)

  #5 outputの最初の列にx~nの値を代入
  output[, 1] <- c(0:n)

  #6 outputのi+1番目の列について計算
  for(i in 1:length(p.vec)){
  output[,i+1] <- BinomDens(x = c(0:n),size = n, prob = p.vec[i])
  }
  return(output)

}
