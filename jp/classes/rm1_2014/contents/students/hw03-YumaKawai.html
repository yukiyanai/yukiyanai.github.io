k_fctrl <- function(n = NULL){# k_fctrl
  ## =n
  ## =x
    if(n == 0){         # n0
        return(1)
    } else if (n < 0){  # n
        return(NA)
    } else if (n > 0){  # n
        x <- n               # nx
        while(n > 1){
            x <- x * (n - 1) # n(n-1)
            n <- n - 1       # (n-1)
        }
        return(x)
    }
}

k_chs <- function(n = NULL, k = NULL){  # k_chs
  output <- k_fctrl(n) / (k_fctrl(k) * k_fctrl(n-k))# 1

k_dbin <- function(x = NULL, size = NULL, prob = NULL){
    ## =x()
    ## size=
  ## prob=
  ## =output()
    output <- rep(NA, length(x)) #
    for(i in 1:length(x)){## i
        output[i] <- k_chs(n = size, k = x[i]) * (prob^x[i]) * (1 - prob)^(size - x[i])
    }
    output <- ifelse(test = is.na(output), yes = 0, no = output) # NA0
    return(output)
}

k_denbin <- function(n = NULL, p.vec = NULL){
    ## =n  (n0n)
    ## p.vec = 
    ## :=output (k)
    
    # 
    output <- matrix(data = NA, nrow = length(0:n), ncol = (length(p.vec)) + 1)
    output[, 1] <- c(0:n)
    for(i in 1:length(p.vec)){
        # outputi+1
        output[,i+1] <- k_dbin(x = c(0:n), size = n, prob = p.vec[i])
    }
    return(output)
}
